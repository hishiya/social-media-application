export const uploadMedia = async (file: File): Promise<string> => {
    // FormData — вбудований браузерний клас для надсилання файлів
    // Це не JSON, а multipart/form-data — спеціальний формат для файлів
    const formData = new FormData();

    // formData.append('field', value) — додаємо поле до форми
    // 'file' — ім'я поля (має співпадати з upload.single('file') на сервері!)
    // file — сам файл типу File (об'єкт браузера)
    formData.append('file', file);

    const response = await axios.post<{ url: string }>(
        `${API_URL}/upload`, // наш новий ендпоінт
        formData,             // надсилаємо FormData (не JSON)
        {
            headers: {
                Authorization: `Bearer ${localStorage.getItem('token')}`,
                // Content-Type НЕ вказуємо — axios виставить його автоматично
                // для FormData він має бути 'multipart/form-data; boundary=...'
                // і axios сам генерує коректний boundary
            },
        }
    );

    catch (err: unknown) {

            // TypeScript не знає тип err, тому перевіряємо структуру вручну
            const message =
                err && typeof err === 'object' && 'response' in err
                    ? (err as { response?: { data?: { message?: string } } }).response?.data?.message
                    : undefined;
